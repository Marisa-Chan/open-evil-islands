   * Types.
     enum cam_record_size = 36

     struct vec3
        float x, y, z

     struct quat
        float w, x, y, z

     struct cam_record
        uint32 unknown1 (time?)
        uint32 unknown2
        vec3 position
        quat rotation

   * Format.
     cam_record records[cam_file_size / cam_record_size]
